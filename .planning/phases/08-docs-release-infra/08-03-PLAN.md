```javascript
// Custom changeset changelog formatter.
// Wraps @changesets/changelog-github and prepends an emoji by bump type.
// Preserves GitHub PR link, commit hash, and user attribution from upstream.
//
// Emoji map (per phase 08 decision D-07):
//   major -> 💥
//   minor -> ✨
//   patch -> 🐞
//
// Root package.json uses "type": "module", so we use a .cjs extension to
// unambiguously load as CommonJS (upstream @changesets/changelog-github
// is published as a dual CJS/ESM module and its CJS build wraps the
// exports under `.default`).

const githubChangelog = require('@changesets/changelog-github').default;

const EMOJI_BY_TYPE = {
  major: '💥',
  minor: '✨',
  patch: '🐞',
};

/** @type {import('@changesets/types').ChangelogFunctions['getReleaseLine']} */
async function getReleaseLine(changeset, type, options) {
  const line = await githubChangelog.getReleaseLine(changeset, type, options);
  const emoji = EMOJI_BY_TYPE[type];
  if (!emoji) return line;
  // Upstream returns a string that begins with "\n\n- " followed by the entry.
  // Inject the emoji right after the leading "- " bullet so the PR link
  // (which comes after the bullet on the same line) stays intact.
  return line.replace(/^(\n\n- )/, `$1${emoji} `);
}

/** @type {import('@changesets/types').ChangelogFunctions['getDependencyReleaseLine']} */
async function getDependencyReleaseLine(changesets, dependenciesUpdated, options) {
  // Delegate unchanged — dependency bumps don't get a top-level emoji.
  return githubChangelog.getDependencyReleaseLine(changesets, dependenciesUpdated, options);
}

module.exports = { getReleaseLine, getDependencyReleaseLine };
```

Notes:
- Use `.default` when requiring `@changesets/changelog-github` (Pitfall 3).
- Use `module.exports = { ... }` — named exports only per project code style.
- File name is `changelog.cjs` (NOT `.js`) so Node unambiguously loads as CommonJS despite root `package.json` `"type": "module"`.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| contributor machine → `pnpm changeset version` execution | The custom `.changeset/changelog.cjs` runs as Node code during the version step, both locally and in CI |
| CI workflow → GitHub API via `@changesets/changelog-github` | The upstream package fetches PR/commit metadata using `GITHUB_TOKEN` |
| Editor (VS Code) → `unpkg.com/@changesets/config/schema.json` | Editor-side schema validation fetches the pinned schema URL |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-08 | Tampering / RCE | `.changeset/changelog.cjs` | mitigate | File is committed to the repo and reviewed on any PR that touches it. The formatter is a pure wrapper around an already-trusted dependency (`@changesets/changelog-github@0.6.0`) — it performs no network I/O, no filesystem writes, no shell execution. It accepts only two inputs (the changeset object and the bump type) and mutates a string. Code review on the introducing PR is the control. ASVS L1 V14. |
| T-08-09 | Supply-chain (upstream compromise) | `@changesets/changelog-github@0.6.0` | accept | Pinned to the exact version already in `pnpm-lock.yaml`. Any future upgrade goes through a standard PR review. The wrapper does not amplify upstream reach. |
| T-08-10 | Information Disclosure | GITHUB_TOKEN via formatter | mitigate | The wrapper delegates PR/commit metadata fetching to upstream — it never logs `options`, never logs the environment, never writes to stderr. Direct inspection of `.changeset/changelog.cjs` confirms no `console.log(options)` or `process.env` references. |
| T-08-11 | Injection (emoji in summary) | `.changeset/changelog.cjs` regex | mitigate | The regex replacement `/^(\n\n- )/` anchors to the start-of-line bullet produced by upstream — it cannot match arbitrary user input in the changeset summary. Even if a user writes `- ` at the start of their summary, the regex only matches the very first two-newline-plus-bullet sequence (the bullet upstream generated), not user-provided content. |
| T-08-12 | Tampering (schema URL DNS) | `unpkg.com` | accept | The `$schema` URL is editor-only — the JSON is validated by changesets' Zod schema at runtime regardless. Even if `unpkg.com` served a malicious schema, the editor's worst case is showing spurious warnings, not code execution. |

Docs updates (Task 2) have no code execution surface — markdown rendering only.
</threat_model>
