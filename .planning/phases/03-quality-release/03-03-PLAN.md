```yaml
# .github/workflows/ci.yml
# Runs on every PR targeting main.
# Pipeline: lint-and-build → unit-test (matrix) → e2e-test
# Per D-09 (full pipeline on every PR), D-10 (ubuntu × Node [20,22] matrix), D-11 (Playwright cache)
name: CI

on:
  pull_request:
    branches: [main]

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

jobs:
  lint-and-build:
    name: Lint & Build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v4
        with:
          version: 10

      - uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: 'pnpm'

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Lint
        run: pnpm lint

      - name: Build
        run: pnpm build

  unit-test:
    name: Unit Tests (Node ${{ matrix.node-version }})
    needs: lint-and-build
    runs-on: ubuntu-latest
    strategy:
      matrix:
        node-version: [20, 22]
    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v4
        with:
          version: 10

      - uses: actions/setup-node@v4
        with:
          node-version: ${{ matrix.node-version }}
          cache: 'pnpm'

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Unit tests with coverage
        run: pnpm test -- --coverage

      - name: Validate package (publint + attw)
        run: pnpm validate

  e2e-test:
    name: E2E Tests (Chromium / Firefox / WebKit)
    needs: lint-and-build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v4
        with:
          version: 10

      - uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: 'pnpm'

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Cache Playwright browsers
        uses: actions/cache@v4
        id: playwright-cache
        with:
          path: ~/.cache/ms-playwright
          key: playwright-${{ hashFiles('**/pnpm-lock.yaml') }}

      - name: Install Playwright browsers (cache miss)
        if: steps.playwright-cache.outputs.cache-hit != 'true'
        run: pnpm exec playwright install --with-deps chromium firefox webkit

      - name: Install Playwright system deps (cache hit)
        if: steps.playwright-cache.outputs.cache-hit == 'true'
        run: pnpm exec playwright install-deps chromium firefox webkit

      - name: Build (needed for E2E fixture to load dist/)
        run: pnpm build

      - name: E2E tests
        run: pnpm test:e2e
```

```yaml
# .github/workflows/release.yml
# Triggers on push to main.
# Uses changesets/action@v1 to:
#   1. Open a "Version PR" (bumps version + generates CHANGELOG.md) when .changeset files are present
#   2. Publish to npm when the Version PR is merged (no .changeset files remain)
# Per D-18 (changesets/action), D-19 (NPM_TOKEN), D-20 (GITHUB_TOKEN auto-provided)
# Requires: NPM_TOKEN secret in GitHub repo → Settings → Secrets and variables → Actions
name: Release

on:
  push:
    branches: [main]

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

permissions:
  contents: write        # needed to push version commits and create git tags
  pull-requests: write   # needed to open Version PR

jobs:
  release:
    name: Release
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0   # required: changesets needs full history to detect changed packages (Pitfall 5)

      - uses: pnpm/action-setup@v4
        with:
          version: 10

      - uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: 'pnpm'
          registry-url: 'https://registry.npmjs.org'   # auto-writes NODE_AUTH_TOKEN to .npmrc

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Build
        run: pnpm build

      - name: Create Version PR or Publish to npm
        id: changesets
        uses: changesets/action@v1
        with:
          publish: pnpm publish --no-git-checks   # --no-git-checks: changesets already committed version bumps
          commit: 'chore: version packages'
          title: 'chore: version packages'
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          NPM_TOKEN: ${{ secrets.NPM_TOKEN }}
          NODE_AUTH_TOKEN: ${{ secrets.NPM_TOKEN }}   # used by .npmrc from registry-url above
```

Note: `--no-git-checks` is required because `changesets/action` commits version bumps before calling publish. Without this flag, pnpm publish would refuse to publish because there are "uncommitted changes" (the version bump commits). This is the standard pattern for changesets + pnpm.

Note: `permissions: contents: write + pull-requests: write` must be declared explicitly because modern GitHub repositories default to read-only token permissions.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CI→npm registry | NPM_TOKEN in GitHub Secrets → pnpm publish. Token must be Automation type (not Publish) to bypass 2FA requirement. |
| CI→GitHub API | GITHUB_TOKEN used by changesets/action to open PRs and push commits. Limited to repository scope. |
| PR branch→main | Only PRs passing the full CI pipeline can merge to main (branch protection rules recommended). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-03-01 | Information Disclosure | NPM_TOKEN exposed in logs | mitigate | Never echo NPM_TOKEN. GitHub Secrets are masked in logs automatically. Use Automation token type (not granular) — narrower blast radius. |
| T-03-03-02 | Elevation of Privilege | Malicious PR gaining write access to main | mitigate | ci.yml runs on pull_request (fork PRs use read-only token). release.yml runs on push to main only — requires branch protection requiring CI to pass. |
| T-03-03-03 | Tampering | changesets/action pushing unsigned version commits | accept | Version commits are mechanical (version bump + changelog). GPG signing adds friction with no material trust benefit for a solo project. |
| T-03-03-04 | Denial of Service | cancel-in-progress: true cancels legitimate CI runs | accept | Only cancels runs for the same PR/ref. New push to a PR always starts fresh run. |
</threat_model>
