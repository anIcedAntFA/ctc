This enforces conventional commit types: feat, fix, chore, docs, refactor, test, ci, perf.

3. Create lefthook.yml per D-13 (pre-commit) and D-14 (pre-push):

```yaml
pre-commit:
  parallel: true
  commands:
    lint:
      run: pnpm biome check --no-errors-on-unmatched {staged_files}
      glob: "*.{ts,js,json}"
    format:
      run: pnpm biome format --no-errors-on-unmatched {staged_files}
      glob: "*.{ts,js,json}"
    typecheck:
      run: pnpm tsc --noEmit
    test:
      run: pnpm vitest run
      glob: "*.{ts,js}"

commit-msg:
  commands:
    commitlint:
      run: pnpm commitlint --edit {1}

pre-push:
  commands:
    build:
      run: pnpm build
    test:
      run: pnpm test
    validate:
      run: pnpm validate
```

Per D-13: pre-commit runs lint, format, typecheck, and unit tests in parallel.
Per D-14: pre-push runs build, test, and validate.
Per D-15: commit-msg runs commitlint.

4. Run `pnpm prepare` (which runs `lefthook install`) to activate the hooks.
